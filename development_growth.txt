namespace = development

# Farmlands (Base Tax)
province_event = {
	id = development.1
	title = development.1.t
	desc = development.1.d
	picture = CITY_DEVELOPMENT_eventPicture
	
	hidden = yes
	
	trigger = {
		is_city = yes
		OR = {
			has_terrain = farmlands
		}
	}
	
	mean_time_to_happen = {
		months = 900
		modifier = {
			factor = 0.5
			has_province_modifier = encouraged_growth_tax
		}
		modifier = {
			factor = 0.95
			province_has_center_of_trade_of_level = 1
		}
		modifier = {
			factor = 0.95
			province_has_center_of_trade_of_level = 2
		}
		modifier = {
			factor = 0.95
			province_has_center_of_trade_of_level = 3
		}
		modifier = {
			factor = 0.95
			development = 10
		}
		modifier = {
			factor = 0.95
			development = 20
		}
		modifier = {
			factor = 0.95
			development = 30
		}
		modifier = {
			factor = 0.95
			development = 40
		}
		modifier = {
			factor = 0.95
			development = 50
		}
		modifier = {
			factor = 0.95
			is_capital = yes
			owner = {
				government_rank = 1
			}
		}
		modifier = {
			factor = 0.95
			is_capital = yes
			owner = {
				government_rank = 2
			}
		}
		modifier = {
			factor = 0.95
			is_capital = yes
			owner = {
				government_rank = 3
			}
		}
		modifier = {
			factor = 10
			has_climate = arctic
		}
		modifier = {
			factor = 0.95
			OR = {
				has_building = temple
				has_building = cathedral
			}
		}
	}
	
	immediate = {
		add_base_tax = 1
		if = {
			limit = {
				owner = {
					has_country_flag = development_display_on
				}
			}
			owner = {
				country_event = {
					id = development_display.1
				}
			}
		}
	}
	
	option = {
		name = development.1.a
	}
}

	id = development.11
	title = development.11.t
	desc = development.11.d
	picture = CITY_DEVELOPMENT_eventPicture
	
	hidden = yes
	
	trigger = {
		is_city = yes
		OR = {
			has_terrain = woods
		}
	}
	
	mean_time_to_happen = {
		months = 1800
		modifier = {
			factor = 0.5
			has_province_modifier = encouraged_growth_production
		}
		modifier = {
			factor = 0.95
			province_has_center_of_trade_of_level = 1
		}
		modifier = {
			factor = 0.95
			province_has_center_of_trade_of_level = 2
		}
		modifier = {
			factor = 0.95
			province_has_center_of_trade_of_level = 3
		}
		modifier = {
			factor = 0.95
			development = 10
		}
		modifier = {
			factor = 0.95
			development = 20
		}
		modifier = {
			factor = 0.95
			development = 30
		}
		modifier = {
			factor = 0.95
			development = 40
		}
		modifier = {
			factor = 0.95
			development = 50
		}
		modifier = {
			factor = 0.95
			is_capital = yes
			owner = {
				government_rank = 1
			}
		}
		modifier = {
			factor = 0.95
			is_capital = yes
			owner = {
				government_rank = 2
			}
		}
		modifier = {
			factor = 0.95
			is_capital = yes
			owner = {
				government_rank = 3
			}
		}
		modifier = {
			factor = 10
			has_climate = arctic
		}
		modifier = {
			factor = 0.95
			OR = {
				has_building = workshop
				has_building = counting_house
			}
		}
	}
	
	immediate = {
		add_base_production = 1
		if = {
			limit = {
				owner = {
					has_country_flag = development_display_on
				}
			}
			owner = {
				country_event = {
					id = development_display.2
				}
			}
		}
	}
	
	option = {
		name = development.11.a
	}
}

			factor = 0.95

		}

		}
		modifier = {
			factor = 0.95
			province_has_center_of_trade_of_level = 3
		}
		modifier = {
			factor = 0.95
			development = 10
		}
		modifier = {
			factor = 0.95
			development = 20
		}
		modifier = {
			factor = 0.95
			development = 30
		}
		modifier = {
			factor = 0.95
			development = 40
		}
		modifier = {
			factor = 0.95
			development = 50
		}
		modifier = {
			factor = 0.95
			is_capital = yes
			owner = {
				government_rank = 1
			}
		}
		modifier = {
			factor = 0.95
			is_capital = yes
			owner = {
				government_rank = 2
			}
		}
		modifier = {
			factor = 0.95
			is_capital = yes
			owner = {
				government_rank = 3
			}
		}
		modifier = {
			factor = 10
			has_climate = arctic
		}
		modifier = {
			factor = 0.95
			OR = {
				has_building = barracks
				has_building = training_fields
			}
